Power BI Implementation Guide · DAX Measures
Core DAX Measures
Copy-paste ready for Power BI Desktop
// Total Revenue
Total Sales =
  SUM(Orders[Sales])

// Profit Margin
Profit Margin % =
  DIVIDE([Total Profit], [Total Sales])

// YoY Growth
YoY Growth % =
  VAR PriorYear =
    CALCULATE([Total Sales],
    SAMEPERIODLASTYEAR('Date'[Date]))
  RETURN DIVIDE([Total Sales]-PriorYear, PriorYear)

// Average Order Value
AOV =
  DIVIDE([Total Sales],
  DISTINCTCOUNT(Orders[Order ID]))

// Customer Lifetime Value
CLV =
  DIVIDE([Total Sales],
  DISTINCTCOUNT(Orders[Customer ID]))

Advanced RFM DAX
Segmentation measures
// Recency Score (1–5)
Recency Score =
  VAR LastPurch =
    CALCULATE(MAX(Orders[Order Date]),
    ALLEXCEPT(Orders,Orders[Customer ID]))
  VAR Days = DATEDIFF(LastPurch, TODAY(), DAY)
  RETURN SWITCH(TRUE(),
    Days <= 30, 5,
    Days <= 90, 4,
    Days <= 180, 3,
    Days <= 365, 2, 1)

// RFM Segment Label
RFM Segment =
  VAR R = [Recency Score]
  VAR F = [Frequency Score]
  VAR M = [Monetary Score]
  RETURN SWITCH(TRUE(),
    R>=4 && F>=4, "Champion",
    R>=3 && F>=3, "Loyal",
    R>=4 && F<=2, "New Customer",
    R<=2 && F>=3, "At Risk",
    R<=2 && F<=2, "Lost", "Average")
